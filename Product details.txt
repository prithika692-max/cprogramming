#include <stdio.h>
#include <stdlib.h>

struct Supplier {
    char name[30];
    char location[30];
};

struct Product {
    int product_id;
    char product_name[30];
    float price;
    struct Supplier sup;  
};

int main() {
    struct Product *ptr;   

    ptr = (struct Product *)malloc(sizeof(struct Product));

    if (ptr == NULL) {
        printf("Memory allocation failed\n");
        return 1;
    }

    printf("Enter Product ID: ");
    scanf("%d", &ptr->product_id);

    printf("Enter Product Name: ");
    scanf("%s", ptr->product_name);

    printf("Enter Product Price: ");
    scanf("%f", &ptr->price);

    printf("Enter Supplier Name: ");
    scanf("%s", ptr->sup.name);

    printf("Enter Supplier Location: ");
    scanf("%s", ptr->sup.location);

    printf("\n--- Product Details ---\n");
    printf("Product ID    : %d\n", ptr->product_id);
    printf("Product Name  : %s\n", ptr->product_name);
    printf("Product Price : %.2f\n", ptr->price);
    printf("Supplier Name : %s\n", ptr->sup.name);
    printf("Location      : %s\n", ptr->sup.location);

    free(ptr);

    return 0;
}
